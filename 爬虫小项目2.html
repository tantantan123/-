def get_one_page(url):
    try:
        headers = {
            'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_13_3) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/65.0.3325.162 Safari/537.36'
        }
        response = requests.get(url, headers=headers)
        if response.status_code == 200:
            return response.text
        return None
    except RequestException
        return None

def parse_one_page(html):
    pattern = re.compile('<dd>.*?board-index.*?>(\d+)</i>.*?data-src="(.*?)".*?name"><a'#非贪婪匹配#
                         + '.*?>(.*?)</a>.*?star">(.*?)</p>.*?releasetime">(.*?)</p>'
                         + '.*?integer">(.*?)</i>.*?fraction">(.*?)</i>.*?</dd>', re.S)#.不可以代表换行符，但是网页中必定不只有一行，应该用re.S来表示整个网页#
    items = re.findall(pattern, html)
    for item in items:
        yield {#yeild指的是生成器#
            'index': item[0],
            'image': item[1],
            'title': item[2],
            'actor': item[3].strip()[3:],
            'time': item[4].strip()[5:],
            'score': item[5] + item[6]
        }

def get_songinfo(html):
    soup=BeautifulSoup(html,'lxml')
    items=soup.find('ul',{'class':'f-hide'}).findAll('li')
    list1=[]
    list2=[]
    list3=[]
    for item in items:
        text=item.find('a').attrs['href'][9:]
        song=item.text
        list1.append(text)
        list2.append(song)
    list3.append(list1)#1为数字#
    list3.append(list2)#2为名字#
    return list3
